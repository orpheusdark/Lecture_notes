SELECT M.MEMBER_NAME, COUNT(A.ATTENDANCE_ID) AS ATTENDANCE_COUNT
FROM MEMBERS M
LEFT JOIN ATTENDANCE A ON M.MEMBER_ID = A.MEMBER_ID
GROUP BY M.MEMBER_NAME
ORDER BY M.MEMBER_NAME;

SELECT MAX(SESSION_COUNT) AS MAX_ATTENDANCE
FROM (
  SELECT MEMBER_ID, COUNT(*) AS SESSION_COUNT
  FROM ATTENDANCE
  GROUP BY MEMBER_ID
);

SELECT M.MEMBER_NAME
FROM MEMBERS M
LEFT JOIN ATTENDANCE A ON M.MEMBER_ID = A.MEMBER_ID
WHERE A.MEMBER_ID IS NULL;
