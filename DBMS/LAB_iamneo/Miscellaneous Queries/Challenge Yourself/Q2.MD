SELECT S.STUDENT_NAME, B.BOOK_TITLE
FROM STUDENTS S
JOIN BORROWINGS B ON S.STUDENT_ID = B.STUDENT_ID
WHERE S.DEPARTMENT = 'CSE';

SELECT S.STUDENT_NAME, B.BOOK_TITLE, B.BORROW_DATE
FROM STUDENTS S
JOIN BORROWINGS B ON S.STUDENT_ID = B.STUDENT_ID
WHERE (S.STUDENT_ID, B.BORROW_DATE) IN (
    SELECT STUDENT_ID, MAX(BORROW_DATE)
    FROM BORROWINGS
    GROUP BY STUDENT_ID
);
