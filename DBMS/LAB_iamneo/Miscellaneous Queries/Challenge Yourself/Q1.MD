SELECT F.FLYER_NAME, SUM(M.MILES_EARNED) AS TOTAL_MILES
FROM FLYERS F
JOIN MILES M ON F.FLYER_ID = M.FLYER_ID
GROUP BY F.FLYER_NAME;

SELECT AVG(MILES_EARNED) AS AVERAGE_MILES_PER_FLIGHT
FROM MILES;

SELECT F.FLYER_NAME
FROM FLYERS F
LEFT JOIN MILES M ON F.FLYER_ID = M.FLYER_ID
WHERE M.FLYER_ID IS NULL;
