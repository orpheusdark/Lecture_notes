SELECT P.PRODUCT_NAME, O.ORDER_ID, O.ORDER_DATE, O.ORDER_STATUS, O.ORDER_QUANTITY
FROM CUSTOMERS C
JOIN PRODUCTS P ON C.CUSTOMER_ID = P.CUSTOMER_ID
JOIN ORDERS O ON C.CUSTOMER_ID = O.CUSTOMER_ID
WHERE C.CUSTOMER_NAME IN ('IVAN BAYROSS', 'MAMTA MUZUMDAR');

SELECT DISTINCT C.CUSTOMER_NAME
FROM CUSTOMERS C
JOIN ORDERS O ON C.CUSTOMER_ID = O.CUSTOMER_ID
WHERE O.ORDER_DATE < TO_DATE('2004-06-01', 'YYYY-MM-DD');
