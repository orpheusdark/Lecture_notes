SELECT SUM(ORDER_QUANTITY * P.SELL_PRICE) AS TOTAL_BILLED 
FROM ORDERS O
JOIN PRODUCTS P ON O.ORDER_ID = P.PRODUCT_ID
WHERE EXTRACT(MONTH FROM O.ORDER_DATE) = 6;
SELECT P.PRODUCT_NAME, O.ORDER_ID, O.ORDER_QUANTITY 
FROM ORDERS O 
JOIN PRODUCTS P ON O.ORDER_ID = P.PRODUCT_ID 
WHERE P.PRODUCT_NAME = 'PULL OVER' AND O.ORDER_QUANTITY < 5;
