SELECT E.EMP_NAME, SUM(C.COMMISSION_AMOUNT) AS TOTAL_COMMISSION
FROM EMPLOYEES E
JOIN COMMISSIONS C ON E.EMP_ID = C.EMP_ID
GROUP BY E.EMP_NAME;

SELECT ROUND(AVG(COMMISSION_AMOUNT), 2) AS AVERAGE_COMMISSION
FROM COMMISSIONS;

SELECT E.EMP_NAME
FROM EMPLOYEES E
LEFT JOIN COMMISSIONS C ON E.EMP_ID = C.EMP_ID
WHERE C.EMP_ID IS NULL;
