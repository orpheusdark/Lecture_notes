SELECT M.TITLE, ROUND(AVG(R.RATING), 2) AS AVG_RATING
FROM MOVIES M
JOIN RATINGS R ON M.MOVIE_ID = R.MOVIE_ID
GROUP BY M.TITLE;

SELECT TITLE, AVG_RATING FROM (
  SELECT M.TITLE, ROUND(AVG(R.RATING), 2) AS AVG_RATING
  FROM MOVIES M
  JOIN RATINGS R ON M.MOVIE_ID = R.MOVIE_ID
  GROUP BY M.TITLE
)
WHERE AVG_RATING = (
  SELECT MAX(AVG_RATING) FROM (
    SELECT ROUND(AVG(R.RATING), 2) AS AVG_RATING
    FROM RATINGS R
    GROUP BY R.MOVIE_ID
  )
);
