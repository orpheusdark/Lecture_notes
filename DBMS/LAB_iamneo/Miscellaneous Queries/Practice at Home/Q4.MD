SELECT U.USERNAME, COUNT(V.VOTE_ID) AS TOTAL_VOTES
FROM USERS U
JOIN VOTES V ON U.USER_ID = V.USER_ID
GROUP BY U.USERNAME;

SELECT USER_ID
FROM VOTES
GROUP BY USER_ID
HAVING COUNT(DISTINCT POLL_ID) > 2;

SELECT U.USERNAME
FROM USERS U
LEFT JOIN VOTES V ON U.USER_ID = V.USER_ID
WHERE V.USER_ID IS NULL;
