DECLARE
  v_n NUMBER;
  v_sum NUMBER;
  
  CURSOR c_numbers IS
    SELECT n FROM numbers;
BEGIN
  FOR rec IN c_numbers LOOP
    v_n := rec.n;
    v_sum := 0;
    
    FOR i IN 1..v_n LOOP
      v_sum := v_sum + i;
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('The sum of numbers from 1 to ' || v_n || ' is: ' || v_sum);
  END LOOP;
END;
/
