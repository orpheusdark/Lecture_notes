DECLARE
  v_num1 NUMBER;
  v_num2 NUMBER;
  v_num3 NUMBER;
  v_max NUMBER;
  
  CURSOR c_numbers IS
    SELECT num1, num2, num3 FROM numbers;
BEGIN
  FOR rec IN c_numbers LOOP
    v_num1 := rec.num1;
    v_num2 := rec.num2;
    v_num3 := rec.num3;
    
    v_max := GREATEST(v_num1, v_num2, v_num3);
    
    DBMS_OUTPUT.PUT_LINE('The maximum number is: ' || v_max);
  END LOOP;
END;
/
