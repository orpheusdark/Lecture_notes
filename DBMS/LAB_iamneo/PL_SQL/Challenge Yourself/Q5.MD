DECLARE 

   v_number NUMBER := 141; 

   v_original_number NUMBER; 

   v_reversed_number NUMBER := 0; 

BEGIN 

   v_original_number := v_number; 

  

   WHILE v_number > 0 LOOP 

      v_reversed_number := v_reversed_number * 10 + MOD(v_number, 10); 

      v_number := FLOOR(v_number / 10); 

   END LOOP;
