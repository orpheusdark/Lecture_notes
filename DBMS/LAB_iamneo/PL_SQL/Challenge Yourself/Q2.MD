DECLARE 

   v_number NUMBER := 17;  -- You can change this value for different inputs 

   v_flag BOOLEAN := TRUE; 

BEGIN 

   IF v_number < 2 THEN 
      v_flag := FALSE;  -- Numbers less than 2 are not prime
   ELSE 
      FOR i IN 2..(v_number/2) LOOP  -- Loop from 2 up to v_number/2
         IF v_number MOD i = 0 THEN  -- If the number is divisible by i, it's not prime
            v_flag := FALSE; 
            EXIT;  -- Exit the loop since it's confirmed not prime
         END IF; 
      END LOOP; 
   END IF;
