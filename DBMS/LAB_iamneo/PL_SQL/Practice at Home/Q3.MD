DECLARE 

   v_string VARCHAR2(100) := 'Hello World from Oracle'; 

   v_word_count NUMBER := 0; 

   v_pos NUMBER := 1; 

BEGIN 

   LOOP 

      v_pos := INSTR(v_string, ' ', v_pos) + 1; 

      EXIT WHEN v_pos = 1; 

      v_word_count := v_word_count + 1; 

   END LOOP; 

  

   v_word_count := v_word_count + 1;  -- For the last word
